#include <Rcpp.h>

using namespace Rcpp;
using std::min_element;
using std::max_element;

// [[Rcpp::export]]
double cpp_min(NumericVector x) {
  return *min_element(x.begin(), x.end()); 
}

// [[Rcpp::export]]
double cpp_max(NumericVector x) {
  return *max_element(x.begin(), x.end()); 
}
r_vector <- c(126, 793, 999, 525, 851, 798, 120, 714, 852, 151,
              190, 41, 612, 826, 507, 105, 14, 237, 669, 7)

cpp_min(r_vector)
#> [1] 7
cpp_max(r_vector)
#> [1] 999

#include <algorithm> // std::min_element, std::max_element
#include <iostream>  // std::cout, std::endl
#include <vector>    // std::vector

int main() {
  // example data
  std::vector<double> cpp_vector {126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                                  190, 41, 612, 826, 507, 105, 14, 237, 669, 7};
  // find minimum value
  auto min_res = std::min_element(std::begin(cpp_vector), std::end(cpp_vector));
  // print
  std::cout << "minimum: " << *min_res << std::endl;
  
  // find maximum value
  auto max_res = std::max_element(std::begin(cpp_vector), std::end(cpp_vector));
  // print
  std::cout << "minimum: " << *max_res << std::endl;
  
  return 0;
}
