#include <iostream>  // std::cout, std::endl
#include <vector>    // std::vector

using std::cout;
using std::endl;
using std::size;

using std::vector;

int main() {
  int array_int[] = {126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                     190, 41, 612, 826, 507, 105, 14, 237, 669, 7};

  // manually calculate `array_int`'s length/size...
  auto n_bytes_of_array_int = sizeof(array_int); // how many bytes used by `array_int`?
  auto n_bytes_of_int = sizeof(int);             // how many bytes used by single `int`?
  auto size_manual = n_bytes_of_array_int / n_bytes_of_int; // divide for result
  cout << size_manual << " ~ calculate `array_int`'s length/size manually" << "\n" << endl;
  
  
  // built-in
  auto res1 = size(array_int);
  std::cout << res1 << " ~ `array_int`'s length/size (built-in)" << std::endl;
  
  vector<double> vec_dbl{126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                         190, 41, 612, 826, 507, 105, 14, 237, 669, 7};
  
  auto res2 = std::size(vec_dbl);
  std::cout << res2 << " ~ `vec_dbl`'s length/size (built-in)" << std::endl;

  return 0;
}

cpp/cpp-min-max.cpp 

#include <algorithm> // std::min_element, std::max_element
#include <iostream>  // std::cout, std::endl
#include <vector>    // std::vector

int main() {
  // example data
  std::vector<double> cpp_vector {126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                                  190, 41, 612, 826, 507, 105, 14, 237, 669, 7};
  // find minimum value
  auto min_res = std::min_element(std::begin(cpp_vector), std::end(cpp_vector));
  // print
  std::cout << "minimum: " << *min_res << std::endl;
  
  // find maximum value
  auto max_res = std::max_element(std::begin(cpp_vector), std::end(cpp_vector));
  // print
  std::cout << "maximum: " << *max_res << std::endl;
  

  return 0;
}

cpp/cpp-unique-h.hpp 

# pragma once

#include <algorithm>       // std::remove_if
#include <unordered_set>   // std::unordered_set
#include <vector>          // std::vector

template <typename T>
auto get_unique(std::vector<T> x) {
  std::unordered_set<T> seen;

  auto find_unique = [&seen] (const T& val) {
    if (seen.find(val) == std::end(seen)) {
      seen.insert(val);
      return false;
    } else {
      return true;
    }
  };

  x.erase(std::remove_if(std::begin(x), std::end(x), find_unique),
          std::end(x));

  return x;
}

def get_unique(x):
  seen = set()
  seen_add = seen.add
  return [i for i in x if not (i in seen or seen_add(i))]

result = get_unique(py_list)

cpp/cpp-sum.cpp 

#include "cpp-sum-h.hpp"

#include <iostream>  // std::cout, std::endl
#include <vector>    // std::vector

using std::cout;
using std::endl;
using std::size;
using std::vector;

int main() {
  vector<double> vec_dbl{126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                         190, 41, 612, 826, 507, 105, 14, 237, 669, 7};
  
  // manually calculate `vec_dbl`'s sum...
  int sum_manual = 0;                           // start at 0
  int n = size(vec_dbl);                        // how many elements are in `vec_dbl`?
  for (int i = 0; i < n; i++) {                 // for each of those elements...
    sum_manual += vec_dbl[i];                   // add them to `sum_manual`
  }
  cout << sum_manual << " ~ sum `vec_dbl` manually" << "\n" << endl; // print the result
  
  // modularize it...
  auto res1 = calc_sum(vec_dbl);
  cout << res1 << " ~ sum `vec_dbl` with template" << "\n" << endl;

  
  vector<int> vec_int{126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                      190, 41, 612, 826, 507, 105, 14, 237, 669, 7};
  auto res2 = calc_sum(vec_int);
  cout << res2 << " ~ sum `vec_int` with template" << "\n" << endl;
  
  
  auto res3 = calc_sum(126, 793, 999, 525, 851, 798, 120, 714, 852, 151, 
                       190, 41, 612, 826, 507, 105, 14, 237, 669, 7);
  cout << res3 << " ~ sum individual numbers passed to variadic template" << "\n" << endl;

  return 0;
}

cpp/cpp-sum-h.hpp 

#pragma once

#include <numeric>   // std::accumulate
#include <vector>    // std::vector

template <typename T>
auto calc_sum(const std::vector<T> &x) { // sum a vector of anything
  T zero = 0;
  return std::accumulate(std::begin(x), std::end(x), zero);
}

template <typename... T>  
auto calc_sum(const T& ...dots) {       // sum variable number of arguments
  return (dots + ...);
}
