What is it?

Given a sequence, which is the middle-most value. Put another way, what value is equally likely to be greater than the first half of a sorted sequence and lesser than the second half of a sorted sequence.

  • Steps
    • sort the sequence
    • if the length of the sequence is odd
      • the middle value is the median
    • if length of the sequence is even
      • the mean of the 2 middle values is the median

Custom R

Note: This is essentially a stripped-down version of median.default().

r_median <- function(x) {
  n <- length(x)
  mid <- (n + 1L) %/% 2L
  partially_sorted <- sort(x, partial = n %/% 2L + 1L)
  if (n %% 2L) {
    return(partially_sorted[mid])
  }      
  ( partially_sorted[[mid]] + partially_sorted[[mid + 1L]] ) / 2L 
}

r_median(r_even_n)
#> [1] 516
r_median(r_odd_n)
#> [1] 507

Custom

Blueprint

using std::begin;
using std::end;
using std::nth_element;
using std::size;

// [[Rcpp::export]]
double cpp_median_blueprint(const NumericVector& x) {
  auto temp = clone(x);
  const auto n = size(x);

  auto mid = begin(temp) + n / 2;
  
  std::nth_element( begin(temp), mid, end(temp) );


  if (n % 2) {
    return *mid;
  }

  const auto mid1 = *mid;
  nth_element( begin(temp), --mid, end(temp) );

  return ( mid1 + *mid ) / 2.0;
}
cpp_median_blueprint(r_even_n)
#> [1] 516
cpp_median_blueprint(r_odd_n)
#> [1] 507

Template Version

template <int RTYPE>
auto cpp_median_(const Vector<RTYPE>& x) {
  auto temp = clone(x);

  const auto n = std::size(temp);

  auto mid = std::begin(temp) + n / 2;

  std::nth_element( std::begin(temp), mid, std::end(temp) );

  if (n % 2) {
    return wrap( *mid );
  }
  
  const auto mid1 = *mid;
  std::nth_element( std::begin(temp), --mid, std::end(temp) );

  return wrap( (mid1 + *mid) / 2.0 );
}

// [[Rcpp::export]]
SEXP cpp_median(const SEXP& x) {
  switch( TYPEOF(x) ) {
    case INTSXP:
      return cpp_median_<INTSXP>(x);
    case REALSXP:
      return cpp_median_<REALSXP>(x);
  }
  warning("Invalid SEXPTYPE %d (%s).\n", TYPEOF(x), type2name(x));
  return R_NilValue;
}

Custom Pure Python

def py_median(x):
  n = len(x)
  sorted_x = sorted(x)
  if n % 2:
    return sorted_x[n // 2]
  return (sorted_x[n // 2 - 1] + sorted_x[n // 2]) / 2
  
res1 = py_median(py_even_n)
res2 = py_median(py_odd_n)

print([res1, res2])
#> [516.0, 507]
