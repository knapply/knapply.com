Define doubler_*() Function

r_doubler <- function(x) {
  x * 2
}
 
def py_doubler(x):
  return x * 2
 
#include <Rcpp.h>

// [[Rcpp::export]]
int cpp_doubler(int x) {
  return x * 2;
}

Run doubler_*() Function

r_doubler(x = 4)
## [1] 8
 
print(py_doubler(x = 4))
## 8
 
cpp_doubler(x = 4)
## [1] 8
